CREATE TABLE `tb_carros` (
	`placa` VARCHAR(10) NOT NULL,
	`marca` VARCHAR(40) NULL DEFAULT NULL,
	`cor` VARCHAR(20) NULL DEFAULT NULL,
	`nome` VARCHAR(30) NULL DEFAULT NULL,
	`ano` VARCHAR(50) NULL DEFAULT NULL,
	`situacao` VARCHAR(10) NULL DEFAULT 'L',
	`valor_reserva` DOUBLE NULL DEFAULT NULL,
	`categoria` VARCHAR(50) NULL DEFAULT NULL,
	`descricao` VARCHAR(120) NULL DEFAULT NULL,
	PRIMARY KEY (`placa`)
)
COLLATE='latin1_swedish_ci'
ENGINE=MyISAM
;
